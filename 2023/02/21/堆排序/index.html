堆的后续遍历（DS排序）

Qustion Description

给定一个整数序列， 把它建成最小堆，输出堆的后序遍历
假定序列无重复数字)

Input Desescription

只有一行，先输入n表示序列包含n个整数，接着输入n个整数。

Output Description

序列转变成最小堆后，输出堆的后序遍历。

输入样例

7 49 3 8 65 97 76 13 27

输出样例

97 76 38 65 49 27 13
Code(C++)
class Node{
public:
    int data;
    int leftchild;
    int rightchild;
    int parent;
    Node()
    {
        data = -1;
        leftchild = -1;
    rightchild = -1;
    parent = -1;
    }
};

void postOrder(Node* a, int pos)
{
    postorder.push(a[pos].data);
    if (a[pos].rightchild != -1)
    {
        postOrder(a, a[pos].rightchild);
        postOrder(a, a[pos].leftchild);
    }
    if(a[pos].rightchild==-1&&a[pos].leftchild!=-1)
     postOrder(a, a[pos].leftchild);
}
void test()
{
    Node* a;
    int n;
    cin >> n;
    a = new Node[n+1];
    for (int i = 1;i <= n;i++)
    {
        if(i/2!=0)
        a[i].parent = i / 2;
        if(2*i<=n)
            a[i].leftchild = 2 * i;
        if (2 * i + 1 <= n)
            a[i].rightchild = 2 * i + 1;
    }
    for (int i = 1;i <= n;i++)
        cin >> a[i].data;
    for (int i = n;i >= 1;i--)
    {
        if (a[i].leftchild!=-1||a[i].rightchild!=-1)
        {
            if (a[i].rightchild!=-1)
            {
                if (a[a[i].leftchild].data < a[a[i].rightchild].data && a[a[i].leftchild].data < a[i].data)
                {
                    swap(a[i].data, a[a[i].leftchild].data);
                    i = n;
                }
                else if (a[a[i].leftchild].data > a[a[i].rightchild].data && a[a[i].rightchild].data < a[i].data)
                {
                    swap(a[i].data, a[a[i].rightchild].data);
                    i = n;
                }
            }
            else
            {
                if (a[a[i].leftchild].data < a[i].data)
                {
                    swap(a[i].data, a[a[i].leftchild].data);
                    i = n;
                }
            }
        }
    }
    postOrder(a, 1);
    print(a, n);

}
int main()
{
test();
}
